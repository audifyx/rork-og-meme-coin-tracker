// theme-selection-settings component
import React, { useState, useEffect } from 'react';
import { ThemeContext } from './theme.context';
import { Box, Button, Dropdown, DropdownItem } from '@shadcn/ui';

const ThemeSelectionSettings = () => {
  const { theme, themes, setTheme } = React.useContext(ThemeContext);
  const [selectedTheme, setSelectedTheme] = useState(theme);

  useEffect(() => {
    setSelectedTheme(theme);
  }, [theme]);

  const handleThemeChange = (themeName: string) => {
    setTheme(themeName);
    setSelectedTheme(themeName);
  };

  return (
    <Box>
      <h2>Theme Selection</h2>
      <p>Select a theme:</p>
      <Dropdown>
        {Object.keys(themes).map((themeName) => (
          <DropdownItem key={themeName}>
            <Button
              onClick={() => handleThemeChange(themeName)}
              variant={selectedTheme === themeName ? 'primary' : 'secondary'}
            >
              {themes[themeName as keyof typeof themes].name}
            </Button>
          </DropdownItem>
        ))}
      </Dropdown>
      <h2>OG Themes</h2>
      <p>Our original themes:</p>
      <ul>
        {Object.keys(themes).map((themeName) => (
          <li key={themeName}>
            <Button
              onClick={() => handleThemeChange(themeName)}
              variant={selectedTheme === themeName ? 'primary' : 'secondary'}
            >
              {themes[themeName as keyof typeof themes].name}
            </Button>
          </li>
        ))}
      </ul>
    </Box>
  );
};

export default ThemeSelectionSettings;
