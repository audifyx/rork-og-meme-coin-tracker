import { createClient } from '@supabase/supabase-js';
import { Database } from './types';

const supabaseUrl = 'https://your-supabase-url.supabase.io';
const supabaseKey = 'your-supabase-key';
const supabaseSecret = 'your-supabase-secret';

const supabase = createClient<Database>(supabaseUrl, supabaseKey, supabaseSecret, {
  schema: 'public',
  autoRefreshToken: true,
  persistSession: true,
});

export const getThemes = async () => {
  try {
    const { data, error } = await supabase
      .from('themes')
      .select('id, name, section, colors');

    if (error) {
      throw error;
    }

    const themes = data as { id: number; name: string; section: string; colors: string };
    return themes;
  } catch (error) {
    console.error('Error fetching themes:', error);
    return [];
  }
};

export const createUserThemePreference = async (userId: number, themeId: number) => {
  try {
    const { data, error } = await supabase
      .from('user_theme_preferences')
      .insert({ user_id: userId, theme_id: themeId });

    if (error) {
      throw error;
    }

    return data as { id: number; user_id: number; theme_id: number };
  } catch (error) {
    console.error('Error creating user theme preference:', error);
    return null;
  }
};

export const getUserThemePreference = async (userId: number) => {
  try {
    const { data, error } = await supabase
      .from('user_theme_preferences')
      .select('theme_id')
      .eq('user_id', userId)
      .limit(1);

    if (error) {
      throw error;
    }

    const userThemePreference = data as { theme_id: number }[];
    return userThemePreference[0];
  } catch (error) {
    console.error('Error fetching user theme preference:', error);
    return null;
  }
};
